package com.clearent.idtech.android.family;


import com.clearent.idtech.android.family.device.VP3300;
import com.clearent.idtech.android.family.device.VP3300Impl;

/**
 * Use this factory to get an object to use for your device.
 */
public class DeviceFactory {

    /**
     * getVP3300 Use this method to get a VP3300 object for interacting with the VivoPay series. Devices support audio jack, audio jack + usb, usb only, and bluetooth.
     *
     * @param applicationContext provide context from your application in order to communicate with the device.
     * @return VP3300 Object
     */
    public static VP3300 getVP3300(ApplicationContext applicationContext) {
        switch (applicationContext.getDeviceType()) {
            case DEVICE_VP3300_AJ:
            case DEVICE_VP3300_AJ_USB:
            case DEVICE_VP3300_USB:
            case DEVICE_VP3300_BT:
                VP3300 vP3300 = new VP3300Impl(applicationContext);
                return vP3300;
            default:
                throw new RuntimeException("No VP3300 device found");
        }
    }
}
