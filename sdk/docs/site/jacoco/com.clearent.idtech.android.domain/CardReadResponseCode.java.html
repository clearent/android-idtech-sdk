package com.clearent.idtech.android.domain;

/**
 * List of responses related to the user interaction with the card reader. If an error is occurring that is not on this list it might be returned
 * in the lcdDisplay method.
 */
public enum CardReadResponseCode {

    USE_CHIP_READER("USE CHIP READER"),
    USE_MAGSTRIPE("USE MAGSTRIPE"),
    TERMINATE("TERMINATE"),
    CONTACTLESS_NOT_SUPPORTED("Contactless not supported"),
    SWIPE_CARD("Swipe card"),
    REMOVE_CARD_AND_TRY_SWIPE("Remove card and try swipe"),
    BAD_SWIPE("Invalid swipe"),
    FALLBACK_TO_SWIPE_FAILED("Waited for the card to be removed before trying swipe. Try again (remove card when prompted)."),
    DECLINED_OFFLINE("Declined Offline"),
    TIMEOUT("Timeout");

    private String displayMessage ;

    CardReadResponseCode(String displayMessage) {
        this.displayMessage = displayMessage;
    }

    public String getDisplayMessage() {
        return displayMessage;
    }

    public void setDisplayMessage(String displayMessage) {
        this.displayMessage = displayMessage;
    }
}
